# 代码实现

一：在 didFinishLaunchingWithOptions 里注册交易的观察者

SKPaymentQueue.default().add(IAPManager.shared) // 需要实现 SKPaymentTransactionObserver

二：发起商品请求，这一步是通过 id 去获取商品的信息

func buy(id: String)  {
     let req = SKProductsRequest.init(productIdentifiers: [id])
     req.delegate = self  //SKProductsRequestDelegate
     req.start()
}

三：在代理方法里，根据商品构建 pay，通过 queue 发起支付

// SKProductsRequestDelegate
func productsRequest(_ request: SKProductsRequest,
                         didReceive response: SKProductsResponse) {
    for prod in response.products {
        let pay = SKMutablePayment.init(product: prod)
        SKPaymentQueue.default().add(pay)
    }
}

四：实现 SKPaymentTransactionObserver

func paymentQueue(_ queue: SKPaymentQueue,
                  updatedTransactions transactions: [SKPaymentTransaction]) {
    for tran in transactions {
        switch tran.transactionState {
        case .purchasing:
            break
        case .purchased:
            // 验证交易
            self.verifyTranscation(tran)
        case .failed:
            self.delegate?.iapManager(self, didComepleteIAP: .fail(tran.error!))
        case .restored:
            // 拿到原始 transaction 再验证
            if let originTran = tran.original {
                self.verifyTranscation(originTran)
            }
        case .deferred:
            break
        }
    }
}

func verifyTranscation(_ tran: SKPaymentTransaction) {
    // 从沙盒拿到 receipt
    // 我这里是发给我们的 server，然后 server 再和苹果验证
    guard let url = Bundle.main.appStoreReceiptURL else { return }
    let data = try? Data.init(contentsOf: url)
    self.checkPurchaseResult(receipt: data?.base64EncodedString() ?? "")
        .then { (apiData) in
            // 验证通过 结束交易
            SKPaymentQueue.default().finishTransaction(tran)
        }
        .catch { (error) in
            // 失败
        }
}

五：处理 restored 的交易。如果一个 purchased 的交易没有执行 finish。那么这个交易再下次 App 启动的时候就会变成 restored。这时我们需要 SKPaymentQueue.default().restoreCompletedTransactions() 这个方法重启交易来完成。这个方法会重新发起一次交易。
